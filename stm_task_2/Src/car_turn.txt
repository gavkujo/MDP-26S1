static inline float wrap180(float a) {
    while (a > 180.0f) a -= 360.0f;
    while (a < -180.0f) a += 360.0f;
    return a;
}

typedef struct {
    MotorCommandType_t type;
    uint32_t left_pwm;
    uint32_t right_pwm;
    uint8_t  valid;
} MotorCommandCache_t;

static MotorCommandCache_t s_last_motor_cmd = {0};

static bool MotorCommand_Send(MotorCommandType_t type,
                              uint32_t left_pwm,
                              uint32_t right_pwm,
                              uint32_t duration_ms)
{
    if (xMotorQueue == NULL) {
        return false;
    }

    MotorCommand_t cmd = {0};
    cmd.cmd_type = type;
    cmd.left_pwm = left_pwm;
    cmd.right_pwm = right_pwm;
    cmd.duration_ms = duration_ms;
    cmd.use_pid = false;
    cmd.target_speed_cms = 0.0f;

    return (osMessageQueuePut(xMotorQueue, &cmd, 0, 0) == osOK);
}

static bool MotorCommand_Ensure(MotorCommandType_t type,
                                uint32_t left_pwm,
                                uint32_t right_pwm)
{
    if (!s_last_motor_cmd.valid ||
        s_last_motor_cmd.type      != type ||
        s_last_motor_cmd.left_pwm  != left_pwm ||
        s_last_motor_cmd.right_pwm != right_pwm) {
        if (MotorCommand_Send(type, left_pwm, right_pwm, 0U)) {
            s_last_motor_cmd.valid     = 1U;
            s_last_motor_cmd.type      = type;
            s_last_motor_cmd.left_pwm  = left_pwm;
            s_last_motor_cmd.right_pwm = right_pwm;
            return true;
        }
        return false;
    }
    return true;
}

static void MotorCommand_RequestBrake(void)
{
    if (MotorCommand_Send(MOTOR_CMD_BRAKE, 0U, 0U, 0U)) {
        s_last_motor_cmd.valid = 0U;
    }
}

static void MotorCommand_RequestStop(void)
{
    if (!s_last_motor_cmd.valid || s_last_motor_cmd.type != MOTOR_CMD_STOP) {
        if (MotorCommand_Send(MOTOR_CMD_STOP, 0U, 0U, 0U)) {
            s_last_motor_cmd.valid     = 1U;
            s_last_motor_cmd.type      = MOTOR_CMD_STOP;
            s_last_motor_cmd.left_pwm  = 0U;
            s_last_motor_cmd.right_pwm = 0U;
        }
    }
}

static void MotorCommand_ResetCache(void)
{
    s_last_motor_cmd.valid = 0U;
}

void Robot_MoveStraight(float target_cm,
                        uint32_t base_pwm_left, uint32_t base_pwm_right)
{
    SharedSensorData_t data;
    (void)target_cm;

    if (!g_motion_goal.active) {
        s_forward_ctx.pwm_posted = 0U;
        s_forward_ctx.brake_sent = 0U;
        return;
    }

    if (!g_motion_goal.started) {
        if (!SharedData_GetSensorData(&data)) {
            return;
        }

        g_motion_goal.start_distance_cm = data.distance_traveled_cm;
        g_motion_goal.start_left_ticks  = data.left_encoder_ticks;
        g_motion_goal.start_right_ticks = data.right_encoder_ticks;
        g_motion_goal.start_yaw_deg     = data.yaw_angle_deg;
        g_motion_goal.started           = 1;
        g_motion_goal.completed         = 0;

        s_forward_ctx.pwm_posted = 0U;
        s_forward_ctx.brake_sent = 0U;
        MotorCommand_RequestStop();
        MotorCommand_ResetCache();
    }

    if (!SharedData_GetSensorData(&data)) {
        return;
    }

    float error_yaw = wrap180(data.yaw_angle_deg - g_motion_goal.start_yaw_deg);

    if (fabsf(error_yaw) > 1.0f) {
        if (error_yaw > 0.0f) {
            Servo_SlightLeft();
        } else {
            Servo_SlightRight();
        }
    } else {
        Servo_Center();
    }

    if (!s_forward_ctx.pwm_posted) {
        if (MotorCommand_Ensure(MOTOR_CMD_FORWARD, base_pwm_left, base_pwm_right)) {
            s_forward_ctx.pwm_posted = 1U;
        }
    }

    float travelled = data.distance_traveled_cm - g_motion_goal.start_distance_cm;
    if (travelled < 0.0f) {
        travelled = 0.0f;
    }

    if (travelled >= g_motion_goal.target_distance_cm) {
        if (!s_forward_ctx.brake_sent) {
            MotorCommand_RequestBrake();
            MotorCommand_RequestStop();
            s_forward_ctx.brake_sent = 1U;
        }

        Servo_Center();
        g_motion_goal.active             = 0;
        g_motion_goal.started            = 0;
        g_motion_goal.completed          = 1;
        g_motion_goal.target_distance_cm = 0.0f;
        current_robot_state              = ROBOT_STATE_IDLE;
        MotorCommand_ResetCache();
        s_forward_ctx.pwm_posted = 0U;
        s_forward_ctx.brake_sent = 0U;
    }
}

void Robot_MoveBackward(float target_cm,
                        uint32_t base_pwm_left, uint32_t base_pwm_right)
{
    SharedSensorData_t data;
    (void)target_cm;

    if (!g_motion_goal.active) {
        s_backward_ctx.pwm_posted = 0U;
        s_backward_ctx.brake_sent = 0U;
        return;
    }

    if (!g_motion_goal.started) {
        if (!SharedData_GetSensorData(&data)) return;

        g_motion_goal.start_distance_cm = data.distance_traveled_cm;
        g_motion_goal.start_left_ticks  = data.left_encoder_ticks;
        g_motion_goal.start_right_ticks = data.right_encoder_ticks;
        g_motion_goal.start_yaw_deg     = data.yaw_angle_deg;
        g_motion_goal.started           = 1;
        g_motion_goal.completed         = 0;

        s_backward_ctx.pwm_posted = 0U;
        s_backward_ctx.brake_sent = 0U;
        MotorCommand_RequestStop();
        MotorCommand_ResetCache();
    }

    if (!SharedData_GetSensorData(&data)) return;

    // yaw correction is inverted when driving backward
    float error_yaw = wrap180(data.yaw_angle_deg - g_motion_goal.start_yaw_deg);

    if (fabsf(error_yaw) > 1.0f) {
        if (error_yaw > 0.0f) {
            // backwards needs opposite steering to keep straight
            Servo_SlightRight();
        } else {
            Servo_SlightLeft();
        }
    } else {
        Servo_Center();
    }

    if (!s_backward_ctx.pwm_posted) {
        if (MotorCommand_Ensure(MOTOR_CMD_BACKWARD, base_pwm_left, base_pwm_right)) {
            s_backward_ctx.pwm_posted = 1U;
        }
    }

    // progress is how much we've moved NEGATIVE from the start (make it positive)
    float moved = g_motion_goal.start_distance_cm - data.distance_traveled_cm;
    if (moved < 0.0f) moved = 0.0f;

    if (moved >= fabsf(g_motion_goal.target_distance_cm)) {
        if (!s_backward_ctx.brake_sent) {
            MotorCommand_RequestBrake();
            MotorCommand_RequestStop();
            s_backward_ctx.brake_sent = 1U;
        }

        Servo_Center();
        g_motion_goal.active             = 0;
        g_motion_goal.started            = 0;
        g_motion_goal.completed          = 1;
        g_motion_goal.target_distance_cm = 0.0f;
        current_robot_state              = ROBOT_STATE_IDLE;
        MotorCommand_ResetCache();
        s_backward_ctx.pwm_posted = 0U;
        s_backward_ctx.brake_sent = 0U;
    }
}

void Car_Turn(float angleDeg)
{
    g_motion_goal.target_distance_cm = 0.0f;
    g_motion_goal.target_angle_deg   = angleDeg;
    g_motion_goal.active             = 1;
    g_motion_goal.started            = 0;
    g_motion_goal.completed          = 0;
    g_motion_goal.ack_sent           = 0;
    g_motion_goal.turn_backward      = 0;

    float yaw_now = IMU_GetYawLatest();               // [0..360)
	g_motion_goal.start_yaw_deg      = yaw_now;
	g_motion_goal.target_abs_yaw_deg = wrap360(yaw_now + angleDeg);
	g_motion_goal.snap_needed        = 0;
	g_motion_goal.snap_active        = 0;

    if (angleDeg < 0.0f) {
        current_robot_state = ROBOT_STATE_TURNING_LEFT;
    } else if (angleDeg > 0.0f) {
        current_robot_state = ROBOT_STATE_TURNING_RIGHT;
    }

    MotorCommand_RequestStop();
    MotorCommand_ResetCache();
}

typedef struct {
    uint8_t  in_use;
    uint8_t  turn_right;       // cached net direction (after backward flip)
    float    target_deg;       // signed target relative to start yaw
    float    tolerance_deg;    // completion tolerance
    float    slow_window_deg;  // window to drop to slow PWM
    uint32_t outer_fast_pwm;
    uint32_t inner_fast_pwm;
    uint32_t outer_slow_pwm;
    uint32_t inner_slow_pwm;
} TurnCtx_t;

static TurnCtx_t s_turn = {0};

static inline float clampf(float value, float lo, float hi)
{
    if (value < lo) return lo;
    if (value > hi) return hi;
    return value;
}

// Forward mapping:
// - Left commands are negative, right commands positive.
// - If turning "backward", left/right sense flips.
// Return true if net desired yaw is to the right (positive).
static bool net_turn_right(bool cmd_left_sign_is_negative, bool backward) {
    bool is_left = cmd_left_sign_is_negative ^ backward;
    return !is_left;
}

#define TURN_TARGET_MIN_DEG          15.0f
#define TURN_TARGET_MAX_DEG         180.0f

#define TURN_TOLERANCE_BASE_DEG       2.0f
#define TURN_TOLERANCE_MIN_DEG        0.8f
#define TURN_TOLERANCE_MAX_DEG        3.0f

#define TURN_SLOW_WINDOW_BASE_DEG    12.0f
#define TURN_SLOW_WINDOW_MIN_DEG      3.0f
#define TURN_SLOW_WINDOW_MAX_DEG     24.0f

#define TURN_PWM_OUTER_FAST_BASE   3300U
#define TURN_PWM_OUTER_SLOW_BASE   2700U
#define TURN_PWM_INNER_FAST_BASE   2800U
#define TURN_PWM_INNER_SLOW_BASE   2400U

#define TURN_PWM_OUTER_FAST_MIN    2400U
#define TURN_PWM_OUTER_SLOW_MIN    2000U
#define TURN_PWM_INNER_FAST_MIN    2000U
#define TURN_PWM_INNER_SLOW_MIN    1600U

#define TURN_PWM_SCALE_MIN           0.6f
#define TURN_PWM_SCALE_MAX           1.3f

#define TURN_BIAS_RIGHT_BASE_DEG     1.0f
#define TURN_BIAS_LEFT_BASE_DEG     -1.0f

static void TurnProfile_Init(TurnCtx_t *profile, float requested_deg, bool turn_right)
{
    float abs_req = fabsf(requested_deg); // still clamp target range only
    abs_req = clampf(abs_req, TURN_TARGET_MIN_DEG, TURN_TARGET_MAX_DEG);

    profile->in_use     = 1U;
    profile->turn_right = turn_right ? 1U : 0U;

    // Apply small fixed bias only for shallow turns (< 60°) to preserve 45° calibration
    float bias = 0.0f;
    if (abs_req <= 60.0f) {
        bias = turn_right ? TURN_BIAS_RIGHT_BASE_DEG : TURN_BIAS_LEFT_BASE_DEG;
    }
    if (abs_req <= 100.0f && abs_req > 60.0f) {
            bias = turn_right ? 0.5f : 3.5f;
        }
    const float signed_target = (turn_right ? abs_req : -abs_req) + bias;
    profile->target_deg = clampf(signed_target, -TURN_TARGET_MAX_DEG, TURN_TARGET_MAX_DEG);

    float tol_scale = 1.0f;
    float slow_scale = 1.0f;
    float pwm_scale = 1.0f;

    if (abs_req >= 175.0f) {
        tol_scale  = 0.25f;
        slow_scale = 0.25f;
        pwm_scale  = 0.60f;
    } else if (abs_req >= 90.0f) {
        tol_scale  = 1.50f;
        slow_scale = 0.50f;
        pwm_scale  = 0.80f;
    }

    profile->tolerance_deg = clampf(TURN_TOLERANCE_BASE_DEG * tol_scale,
                                     TURN_TOLERANCE_MIN_DEG,
                                     TURN_TOLERANCE_MAX_DEG);

    profile->slow_window_deg = clampf(TURN_SLOW_WINDOW_BASE_DEG * slow_scale,
                                      TURN_SLOW_WINDOW_MIN_DEG,
                                      TURN_SLOW_WINDOW_MAX_DEG);

    uint32_t outer_fast = (uint32_t)(TURN_PWM_OUTER_FAST_BASE * pwm_scale + 0.5f);
    uint32_t outer_slow = (uint32_t)(TURN_PWM_OUTER_SLOW_BASE * pwm_scale + 0.5f);
    uint32_t inner_fast = (uint32_t)(TURN_PWM_INNER_FAST_BASE * pwm_scale + 0.5f);
    uint32_t inner_slow = (uint32_t)(TURN_PWM_INNER_SLOW_BASE * pwm_scale + 0.5f);

    if (outer_fast < TURN_PWM_OUTER_FAST_MIN) outer_fast = TURN_PWM_OUTER_FAST_MIN;
    if (outer_slow < TURN_PWM_OUTER_SLOW_MIN) outer_slow = TURN_PWM_OUTER_SLOW_MIN;
    if (inner_fast < TURN_PWM_INNER_FAST_MIN) inner_fast = TURN_PWM_INNER_FAST_MIN;
    if (inner_slow < TURN_PWM_INNER_SLOW_MIN) inner_slow = TURN_PWM_INNER_SLOW_MIN;

    profile->outer_fast_pwm = outer_fast;
    profile->outer_slow_pwm = outer_slow;
    profile->inner_fast_pwm = inner_fast;
    profile->inner_slow_pwm = inner_slow;
}

static float s_prev_rem = 0.0f;
static bool  s_prev_rem_valid = false;

void Robot_TurnUpdate(void)
{
    if (!g_motion_goal.active) {
        MotorCommand_RequestStop();
        s_prev_rem_valid = false;
        s_turn = (TurnCtx_t){0};
        return;
    }
    if (fabsf(g_motion_goal.target_angle_deg) < 0.001f) {
        MotorCommand_RequestStop();
        g_motion_goal.active = 0;
        s_turn = (TurnCtx_t){0};
        return;
    }

    SharedSensorData_t sd;
    if (!SharedData_GetSensorData(&sd)) {
        MotorCommand_RequestStop();
        return;
    }

    const bool cmd_left_sign = (g_motion_goal.target_angle_deg < 0.0f);
    const bool backward      = (g_motion_goal.turn_backward != 0);
    const bool net_right     = net_turn_right(cmd_left_sign, backward);

    if (!g_motion_goal.started) {
        g_motion_goal.start_yaw_deg = sd.yaw_angle_deg;
        s_prev_rem_valid = false;

        TurnProfile_Init(&s_turn, g_motion_goal.target_angle_deg, net_right);
        g_motion_goal.started   = 1U;
        g_motion_goal.completed = 0U;

        if (net_right ^ backward) Servo_Right();
        else                      Servo_Left();

        MotorCommand_RequestStop();
        MotorCommand_ResetCache();
        current_robot_state = net_right ? ROBOT_STATE_TURNING_RIGHT : ROBOT_STATE_TURNING_LEFT;
        return;
    }

    const float delta = wrap180(sd.yaw_angle_deg - g_motion_goal.start_yaw_deg);
    const float tgt   = s_turn.target_deg;
    const float rem   = tgt - delta;

    const bool crossed = s_prev_rem_valid &&
                         ((rem > 0.0f && s_prev_rem < 0.0f) ||
                          (rem < 0.0f && s_prev_rem > 0.0f));

    if (fabsf(rem) <= s_turn.tolerance_deg || crossed) {
        MotorCommand_RequestBrake();
        MotorCommand_RequestStop();
        Servo_Center();

        g_motion_goal.active           = 0U;
        g_motion_goal.started          = 0U;
        g_motion_goal.completed        = 1U;

        // NEW: ask for a snap pass using the absolute target we computed earlier
        //g_motion_goal.snap_needed      = 1U;
        // leave target_abs_yaw_deg as-is for the snap block to use
        if (fabsf(g_motion_goal.target_angle_deg) > 50.0f) {
        	g_motion_goal.snap_needed = 1U;
        }

        g_motion_goal.target_angle_deg = 0.0f;
        g_motion_goal.turn_backward    = 0U;
        current_robot_state            = ROBOT_STATE_IDLE;
        s_prev_rem_valid               = false;
        s_turn = (TurnCtx_t){0};
        return;
    }


    const bool phase_forward = !backward;
    const bool turn_right    = s_turn.in_use ? (s_turn.turn_right != 0U) : net_right;

    uint32_t outer_fast = s_turn.outer_fast_pwm;
    uint32_t inner_fast = s_turn.inner_fast_pwm;
    uint32_t outer_slow = s_turn.outer_slow_pwm;
    uint32_t inner_slow = s_turn.inner_slow_pwm;

    if (turn_right) {
        outer_fast += 50U;
        outer_slow += 40U;
    } else {
        outer_fast -= 30U;
        outer_slow -= 40U;
    }

    const bool slow_zone = (fabsf(rem) <= s_turn.slow_window_deg);
    const uint32_t pwm_outer = slow_zone ? outer_slow : outer_fast;
    const uint32_t pwm_inner = slow_zone ? inner_slow : inner_fast;

    uint32_t left_pwm = 0U, right_pwm = 0U;
    if (phase_forward) {
        if (turn_right) {
            left_pwm  = pwm_outer;
            right_pwm = pwm_inner;
            MotorCommand_Ensure(MOTOR_CMD_FORWARD, left_pwm, right_pwm);
        } else {
            left_pwm  = pwm_inner;
            right_pwm = pwm_outer;
            MotorCommand_Ensure(MOTOR_CMD_FORWARD, left_pwm, right_pwm);
        }
    } else {
        if (turn_right) {
            left_pwm  = pwm_outer;
            right_pwm = pwm_inner;
            MotorCommand_Ensure(MOTOR_CMD_BACKWARD, left_pwm, right_pwm);
        } else {
            left_pwm  = pwm_inner;
            right_pwm = pwm_outer;
            MotorCommand_Ensure(MOTOR_CMD_BACKWARD, left_pwm, right_pwm);
        }
    }

    s_prev_rem = rem;
    s_prev_rem_valid = true;
}
