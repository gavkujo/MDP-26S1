static inline bool IR_ReadFast(float* left_cm, float* right_cm)
{

  float r = IR2_GetDistanceCm();

/* Evaluate a boolean predicate (valid/invalid) for the selected sensor(s). */
typedef bool (*IRPred_t)(float d_cm);

static bool IR_SelectAndTest(IRWhich_t which, IRPred_t pred, float l, float r)
{
  switch (which) {
    case IR_LEFT:   return pred(l);
    case IR_RIGHT:  return pred(r);
    case IR_EITHER: return pred(l) || pred(r);
    case IR_BOTH:   return pred(l) && pred(r);
    default:        return false;
  }
}

    g_motion_goal.active             = 0;
    g_motion_goal.started            = 0;

    g_motion_goal.ack_sent           = 0;
    g_motion_goal.target_distance_cm = 0.0f;

static inline void Goal_StartForward(float cm) {
    Goal_Clear();
    current_robot_state              = ROBOT_STATE_MOVING_FORWARD;
    g_motion_goal.target_distance_cm = (cm >= 0.0f) ? cm : -cm;  // positive = forward
    g_motion_goal.active             = 1;
}

static inline void Goal_StartBackward(float cm) {
    Goal_Clear();
    current_robot_state              = ROBOT_STATE_MOVING_BACKWARD;
    g_motion_goal.target_distance_cm = -((cm >= 0.0f) ? cm : -cm);  // negative = backward
    g_motion_goal.active             = 1;
}

static inline void Goal_StartTurnDeg(float deg_signed /* +right, -left */) {
    Goal_Clear();
    g_motion_goal.target_angle_deg = deg_signed;
    g_motion_goal.active           = 1;
    float yaw_now = IMU_GetYawLatest();            // [0..360)
	g_motion_goal.start_yaw_deg     = yaw_now;
	g_motion_goal.target_abs_yaw_deg = wrap360(yaw_now + deg_signed);

	g_motion_goal.snap_needed = 0;
	g_motion_goal.snap_active = 0;
    current_robot_state =
        (deg_signed < 0.0f) ? ROBOT_STATE_TURNING_LEFT :
        (deg_signed > 0.0f) ? ROBOT_STATE_TURNING_RIGHT :
                              ROBOT_STATE_IDLE;
}

/* ===== IR drive via the goal-runner (uses Robot_Move* for servo correction) ===== */
static inline bool ir_in_range(float d_cm, float min_cm, float max_cm)
{
  if (d_cm < 0.0f) return false;        // invalid reading is never "in"
  if (min_cm > max_cm) { float t=min_cm; min_cm=max_cm; max_cm=t; }
  return (d_cm >= min_cm) && (d_cm <= max_cm);
}

static inline void IRDrive_Ensure(bool forward)
{
  /* If no active goal, arm a large-distance drive so ControlTask moves it each tick */
  if (!g_motion_goal.active) {
    if (forward) {
      Goal_StartForward(10000.0f);
    } else {
      Goal_StartBackward(10000.0f);
    }
  } else {
    /* If an angle goal sneaks in, clear it and re-arm distance drive */
    if (fabsf(g_motion_goal.target_angle_deg) > 0.0f) {
      Goal_Clear();
      if (forward) Goal_StartForward(10000.0f);
      else         Goal_StartBackward(10000.0f);
    }
  }
}

static inline void IRDrive_StopNow(void)
{
  Motors_Brake();
  osDelay(200);
  Motors_Stop();
  Servo_Center();
  Goal_Clear();
  current_robot_state = ROBOT_STATE_IDLE;
}

/* Shared finish routine: brake → stop → center → mark goal completed. */
static void IR_StopAndComplete(void)
{
  IRDrive_StopNow();  // ensure the goal-driven motion is halted
  g_motion_goal.completed = 1;
  //g_motion_goal.snap_needed = 1;  // ask for snap pass
}

/* Generic runner used by the helpers below.  (with arming + failsafe) */
static bool IRStraightCore(bool forward,
                           IRWhich_t which,
                           IRPred_t stop_when /* valid/invalid */,
                           float min_cm, float max_cm,
                           bool use_band_for_valid)
{
  const uint16_t PWM_L = 2750U, PWM_R = 2750U;   // (PWM via goal-runner)
  const uint8_t  HIT_COUNT = 2;                // consecutive frames to confirm
  const uint8_t  ARM_TICKS = 4;                // ~40ms at 10ms loop before allowing stop (faster)
  const uint16_t FAILSAFE_TICKS = 300;         // ~3s max in this step

  static uint8_t  hits_fwd_invalid = 0, hits_fwd_valid = 0;
  static uint8_t  hits_back_invalid = 0, hits_back_valid = 0;

  static uint8_t  arm_fwd_invalid = 0, arm_fwd_valid = 0;
  static uint8_t  arm_back_invalid = 0, arm_back_valid = 0;

  static uint16_t fs_fwd_invalid = 0, fs_fwd_valid = 0;
  static uint16_t fs_back_invalid = 0, fs_back_valid = 0;

  uint8_t  *hits_ptr, *arm_ptr;
  uint16_t *fs_ptr;

  if (forward) {
    if (stop_when == ir_is_invalid) { hits_ptr=&hits_fwd_invalid; arm_ptr=&arm_fwd_invalid; fs_ptr=&fs_fwd_invalid; }
    else                             { hits_ptr=&hits_fwd_valid;   arm_ptr=&arm_fwd_valid;   fs_ptr=&fs_fwd_valid;   }
  } else {
    if (stop_when == ir_is_invalid) { hits_ptr=&hits_back_invalid; arm_ptr=&arm_back_invalid; fs_ptr=&fs_back_invalid; }
    else                             { hits_ptr=&hits_back_valid;   arm_ptr=&arm_back_valid;   fs_ptr=&fs_back_valid;   }
  }

  float L = -1.0f, R = -1.0f;

  /* Ensure the ControlTask is actually driving using your goal-based path */
  IRDrive_Ensure(forward);

  /* Read sensors NOW (don’t wait for timer frame) */
  if (!IR_ReadFast(&L, &R)) {
    if (*arm_ptr < ARM_TICKS) (*arm_ptr)++;
    if (*fs_ptr  < FAILSAFE_TICKS) (*fs_ptr)++;
    return false;
  }

  /* Arming window: move first before we allow stopping */
  if (*arm_ptr < ARM_TICKS) {
    (*arm_ptr)++;
    *hits_ptr = 0;
    if (*fs_ptr  < FAILSAFE_TICKS) (*fs_ptr)++;
    return false;
  }

  /* Stop predicate */
  bool pass = false;

  if (stop_when == ir_is_valid && use_band_for_valid) {
    // Only the "valid" case uses the band; "invalid" behavior is unchanged
    bool Lpass = ir_in_range(L, min_cm, max_cm);
    bool Rpass = ir_in_range(R, min_cm, max_cm);
    switch (which) {
      case IR_LEFT:   pass = Lpass;           break;
      case IR_RIGHT:  pass = Rpass;           break;
      case IR_EITHER: pass = Lpass || Rpass;  break;
      case IR_BOTH:   pass = Lpass && Rpass;  break;
      default:        pass = false;           break;
    }
  } else {
    // Original behavior (valid = >0, invalid = <0)
    pass = IR_SelectAndTest(which, stop_when, L, R);
  }

  if (pass) {
    if (++(*hits_ptr) >= HIT_COUNT) {
      IR_StopAndComplete();
      *hits_ptr = 0; *arm_ptr = 0; *fs_ptr = 0;
      return true;
    }
  } else {
    *hits_ptr = 0;
  }


  /* Failsafe so we don't hang forever */
  if (++(*fs_ptr) >= FAILSAFE_TICKS) {
    IRDrive_StopNow();
    *hits_ptr = 0; *arm_ptr = 0; *fs_ptr = 0;
    return true;  // treat as completed so the script advances
  }

  (void)PWM_L; (void)PWM_R; // kept for clarity; actual PWM used in your runner
  return false;
}

/* Simple "go straight then stop at X cm" */
static bool UltrasonicStop_StraightSimple(float stop_cm)
{
  const uint16_t PWM_L = 2750U, PWM_R = 2750U;
  const float    STOP_GUARD_CM = 5.0f;
  const uint8_t  HIT_COUNT   = 2;
  const uint16_t BRAKE_DWELL_MS = 1250U;

  static uint8_t hits = 0;

  SharedSensorData_t sd;
  if (!SharedData_GetSensorData(&sd)) {
    Robot_MoveStraight(10000.0f, PWM_L, PWM_R);
    return false;
  }

  Robot_MoveStraight(10000.0f, PWM_L, PWM_R);

  if (sd.front_distance_cm > 0.0f) {
    if (sd.front_distance_cm <= (stop_cm + STOP_GUARD_CM)) {
      if (++hits >= HIT_COUNT) {
        Motors_Brake();
        osDelay(BRAKE_DWELL_MS);
        Motors_Stop();
        Servo_Center();

        g_motion_goal.active             = 0;
        g_motion_goal.started            = 0;
        g_motion_goal.completed          = 1;
        g_motion_goal.target_distance_cm = 0.0f;
        current_robot_state              = ROBOT_STATE_IDLE;

        hits = 0;
        return true;
      }
    } else {
      hits = 0;
    }
  }

  return false;
}


/* USER CODE END Header_CommunicationTask */
void CommunicationTask(void *argument)
{
  /* USER CODE BEGIN CommunicationTask */
  HAL_UART_Receive_IT(&huart3, &uart_rx_buffer[0], 1);

  for(;;)
  {
    if (s_uart_char_ready) {
      char single = s_uart_single_char;
      s_uart_char_ready = false;

      if (single == 's' || single == 'S') {
        if (!s_script_mode_enabled) {
          s_script_mode_enabled = true;
          s_script_start_requested = true;
          osMessageQueueReset(xCommandQueue);
          SendACK("SCRIPT_START");
        } else {
          SendACK("SCRIPT_BUSY");
        }
      } else {
        if (single == 'l' || single == 'L') single = 'L';
        else if (single == 'r' || single == 'R') single = 'R';

        if (single == 'L' || single == 'R') {
          if (!s_script_mode_enabled) {
            SendACK("WAIT_FOR_S");
          } else if (s_detection_request != DETECT_REQ_NONE) {
            s_detection_value = single;
            s_detection_available = true;
            SendACK("DETECTION_RX");
          } else {
            SendACK("DETECTION_IGNORED");
          }
        }
      }
    }

    if (uart_rx_index >= 7) {
      uart_cmd_buffer[7] = '\0';
      uart_rx_index = 0;
      memset(uart_cmd_buffer, 0, sizeof(uart_cmd_buffer));
      if (s_script_mode_enabled) {
        SendACK("SCRIPT_MODE_ONLY");
      } else {
        SendACK("WAIT_FOR_S");
      }
    }

    osDelay(10);
  }
  /* USER CODE END CommunicationTask */
}

              STEP_START_FWD(PHASE_D_INCH_FORWARD_CM);
        	  //s_step++;
            break;

          case 11: /* Forward until right IR loses obstacle; record width */
            if (!s_phaseD_run2_zeroed) {
              SharedData_ResetDistance();
              s_phaseD_run2_zeroed = true;
            }
            if (IRStop_Forward_UntilInvalid_Right()) {
              s_obstacle2_width_cm = clamp_nonnegative(GetDistanceTravelled());
              s_phaseD_run2_zeroed = false;
              g_motion_goal.completed = 0;  // Clear completed flag to prevent step skip
              s_step++;
            }
            break;

          case 12: /* Final right 90 to face corridor again */
            if (!g_motion_goal.active && g_motion_goal.completed) {
              g_motion_goal.completed = 0;
              SharedData_ResetDistance();
              PHASE_GOTO(PH_E);
              break;
            }
            STEP_TURN_R90();
            break;

          default: break;
        }
      } break;

      /* ===================== D_R: second obstacle (det=R) ===================== */
      case PH_D_R: {
        /* Reset all motion state at phase entry to prevent carryover bugs */
        static bool case1_entered = false;

        if (!s_started) {
          s_started = 1;
          g_motion_goal.completed = 0;
          g_motion_goal.snap_needed = 0;
          g_motion_goal.snap_active = 0;
          case1_entered = false;
        }

        switch (s_step) {
          case 0: /* Turn right 90 */
            STEP_TURN_R90();
            break;

          case 1: /* Reverse until left IR valid */
            if (!case1_entered) {
              case1_entered = true;
              // Ensure clean state: stop any residual motion, clear goal flags
              Motors_Stop();
              Goal_Clear();
              osDelay(100);  // Brief settle
              SendACK("D_R:CASE_1_START");
            }
            if (IRStop_Backward_UntilValid_Left(IR_VALID_MIN_CM, IR_VALID_MAX_CM)) {
              s_phaseD_run1_zeroed = false;
              g_motion_goal.completed = 0;  // Clear completed flag to prevent step skip
              case1_entered = false;  // Reset for potential retry
              s_step++;
              SendACK("D_R:CASE_1_DONE");
            }
            break;

          case 2: /* Inch forward */
              //STEP_START_FWD(PHASE_D_INCH_FORWARD_CM);
        	  s_step++;
            break;

          case 3: /* Forward until left IR loses obstacle */
            if (!s_phaseD_run1_zeroed) {
              SharedData_ResetDistance();
              s_phaseD_run1_zeroed = true;
            }
            if (IRStop_Forward_UntilInvalid_Left()) {
              s_phaseD_run1_zeroed = false;
              g_motion_goal.completed = 0;  // Clear completed flag to prevent step skip
              s_step++;
            }
            break;

          case 4: /* First half 180° (left 90) */
            STEP_TURN_L90();
            break;

          case 5: /* After first 90, reverse to align left IR */
            if (IRStop_Backward_UntilValid_Left(IR_VALID_MIN_CM, IR_VALID_MAX_CM)) {
              g_motion_goal.completed = 0;
              s_step++;
              SendACK("D_R:ALIGN_IR_REVERSE");
            }
            break;

          case 6: /* Forward until left IR loses */
            if (IRStop_Forward_UntilInvalid_Left()) {
              g_motion_goal.completed = 0;
              s_step++;
              SendACK("D_R:ALIGN_IR_FORWARD");
            }
            break;

          case 7:
        	  STEP_START_FWD(19.0f);
        	  break;

          case 8: /* Second half 180° (left 90) */
            STEP_TURN_L90();
            break;

          case 9: /* Reverse until left IR Invalid */
            if (IRStop_Backward_UntilInvalid_Left()) {
              s_phaseD_run2_zeroed = false;
              g_motion_goal.completed = 0;  // Clear completed flag to prevent step skip
              s_step++;
            }
            break;

          case 10: /* Inch forward */
            STEP_START_FWD(PHASE_D_INCH_FORWARD_CM);
        	//s_step++;
            break;

          case 11: /* Forward until left IR loses obstacle; capture width */
            if (!s_phaseD_run2_zeroed) {
              SharedData_ResetDistance();
              s_phaseD_run2_zeroed = true;
            }
            if (IRStop_Forward_UntilInvalid_Left()) {
              s_obstacle2_width_cm = clamp_nonnegative(GetDistanceTravelled());
              s_phaseD_run2_zeroed = false;
              g_motion_goal.completed = 0;  // Clear completed flag to prevent step skip
              s_step++;
            }
            break;

          case 12: /* Final left 90 */
            if (!g_motion_goal.active && g_motion_goal.completed) {
              g_motion_goal.completed = 0;
              SharedData_ResetDistance();
              PHASE_GOTO(PH_E);
              break;
            }
            STEP_TURN_L90();
            break;

          default: break;
        }
      } break;

      /* ===================== E ===================== */
            case PH_E: {
              /* Reset all motion state at phase entry to prevent carryover bugs */
              static bool sidestep_done = false;  // NEW: Flag for IR-based sidestep sub-steps

              if (!s_started) {
                s_started = 1;
                g_motion_goal.completed = 0;
                g_motion_goal.snap_needed = 0;
                g_motion_goal.snap_active = 0;
                s_phaseE_final_run_started = false;
                sidestep_done = false;  // NEW: Reset flag
              }

              switch (s_step) {
                case 0: { /* Go straight the recorded part2 distance */
                  float target = clamp_nonnegative(s_part2_cm);
                  if (target <= 0.0f) { s_step++; break; }
                  if (s_phaseC_detection == 'L') STEP_START_FWD(target + 45.0f);
                  else STEP_START_FWD(target + 70.0f);

                } break;

//                case 1: { /* Inch forward 20 cm */
//                  STEP_START_FWD(PHASE_E_INCH_FORWARD_CM);
//                } break;

                case 1: /* First 90 (opposite of detection) */
                  if (s_phaseC_detection == 'R') STEP_TURN_L90();
                  else                           STEP_TURN_R90();
                  break;

                case 2: /* IR-based sidestep: forward until IR valid, then backward until invalid */
                  if (!sidestep_done) {
                    // Move forward until appropriate IR becomes valid
                    bool ir_valid_reached = false;
                    if (s_phaseC_detection == 'R') {
                      ir_valid_reached = IRStop_Forward_UntilValid_Left(IR_VALID_MIN_CM, IR_VALID_MAX_CM);
                    } else {
                      ir_valid_reached = IRStop_Forward_UntilValid_Right(IR_VALID_MIN_CM, IR_VALID_MAX_CM);
                    }
                    if (ir_valid_reached) {
                      sidestep_done = true;
                      g_motion_goal.completed = 0;  // Reset for next sub-step
                    }
                  } else {
                    // Inch backwards fixed amt
                	  if (s_phaseC_detection == 'L') STEP_START_BACK(7.0f);
                	  else STEP_START_BACK(7.0f);
//                    bool ir_invalid_reached = false;
//                    if (s_phaseC_detection == 'R') {
//                      ir_invalid_reached = IRStop_Backward_UntilInvalid_Left();
//                    } else {
//                      ir_invalid_reached = IRStop_Backward_UntilInvalid_Right();
//                    }
//                    if (ir_invalid_reached) {
//                      sidestep_done = false;
//                      g_motion_goal.completed = 0;  // Reset
//                      s_step++;  // Advance to case 4 (second 90° turn)
//                    }
                  }
                  break;
//                case 4:
//                	if (s_phaseC_detection == 'R') STEP_START_BACK(10.0f);
//				    else                           STEP_START_FWD(10.0f);
//
//                	break;

                case 3: /* Second 90 to face corridor again */
                  if (s_phaseC_detection == 'R') STEP_TURN_R90();
                  else                           STEP_TURN_L90();
                  break;

                case 4: /* Final straight until ultrasonic stop */
                  if (!s_phaseE_final_run_started) {
                    s_phaseE_final_run_started = true;
                    Goal_StartForward(10000.0f);
                  }
                  if (UltrasonicStop_StraightSimple(PHASE_E_FINAL_STOP_CM)) {
                    s_phaseE_final_run_started = false;
                    PHASE_GOTO(PH_Z);
                  }
                  break;

                default: break;
              }
            } break;




#if 0
      /* Legacy phases retained for reference */
      case PH_F:
      case PH_G_L:
      case PH_G_R:
      case PH_H:
        break;
#endif

      case PH_Z:
      default:
        script_active = false;
        s_script_mode_enabled = false;
        s_detection_request = DETECT_REQ_NONE;
        s_detection_available = false;
        s_detection_value = '\0';
        s_phaseD_run1_zeroed = false;
        s_phaseD_run2_zeroed = false;
        s_phaseE_final_run_started = false;
        SendACK("SCRIPT_DONE");
        break;
    }

  }
  /* USER CODE END ControlTask */
}

/**
* @brief Function implementing the MotorTask thread.
* @param argument: Not used
* @retval None
*/
/* USER CODE END Header_MotorTask */
void MotorTask(void *argument)
{
  /* USER CODE BEGIN MotorTask */
  MotorCommand_t motor_cmd;
  osStatus_t status;

  for(;;)
  {
    status = osMessageQueueGet(xMotorQueue, &motor_cmd, NULL, 100);
    if (status == osOK) {
      switch (motor_cmd.cmd_type) {
        case MOTOR_CMD_STOP:
          Motors_Stop();
          break;

        case MOTOR_CMD_FORWARD:
          Motors_ForwardCounts(motor_cmd.left_pwm, motor_cmd.right_pwm);
          if (motor_cmd.duration_ms > 0) {
            osDelay(motor_cmd.duration_ms);
            Motors_Brake();
            Motors_Stop();
          }
          break;

        case MOTOR_CMD_BACKWARD:
          Motors_ReverseCounts(motor_cmd.left_pwm, motor_cmd.right_pwm);
          if(motor_cmd.duration_ms > 0){
        	  osDelay(motor_cmd.duration_ms);
        	  Motors_Brake();
        	  Motors_Stop();
          }
          break;

        case MOTOR_CMD_BRAKE:
          Motors_Brake();
          break;

        default:
          Motors_Stop();
          break;
      }

      if (motor_cmd.servo_angle > 0) {
        // Servo_SetPosition(motor_cmd.servo_angle);
      }
    }

    osDelay(10);
  }
  /* USER CODE END MotorTask */
}

/**
 * @brief Process robot command from RPi
 */
void ProcessRobotCommand(RobotCommand_t* cmd) {
  MotorCommand_t motor_cmd;

  if (!cmd) return;

  SharedData_UpdateDetection(cmd->detection_data);
  memset(&motor_cmd, 0, sizeof(motor_cmd));

  switch (cmd->cmd_type) {
    case CMD_FORWARD:
      current_robot_state = ROBOT_STATE_MOVING_FORWARD;
      g_motion_goal.target_distance_cm = (float)cmd->units;
      g_motion_goal.target_angle_deg = 0.0f;
      g_motion_goal.active = 1;
      g_motion_goal.started = 0;
      g_motion_goal.completed = 0;
      g_motion_goal.ack_sent = 0;
      g_motion_goal.start_distance_cm = 0.0f;
      g_motion_goal.start_yaw_deg = 0.0f;
      g_motion_goal.start_left_ticks = 0;
      g_motion_goal.start_right_ticks = 0;
      command_in_progress = true;
      break;

    case CMD_BACKWARD:
      current_robot_state = ROBOT_STATE_MOVING_BACKWARD;
      g_motion_goal.target_distance_cm = -(float)cmd->units;
      g_motion_goal.target_angle_deg   = 0.0f;
      g_motion_goal.active = 1;
      g_motion_goal.started = 0;
      g_motion_goal.completed = 0;
      g_motion_goal.ack_sent = 0;
      g_motion_goal.start_distance_cm = 0.0f;
      g_motion_goal.start_yaw_deg = 0.0f;
      g_motion_goal.start_left_ticks = 0;
      g_motion_goal.start_right_ticks = 0;
      g_motion_goal.turn_backward = 0;
      command_in_progress = true;
      break;

    case CMD_STOP:
      current_robot_state = ROBOT_STATE_IDLE;
      motor_cmd.cmd_type = MOTOR_CMD_STOP;
      osMessageQueuePut(xMotorQueue, &motor_cmd, 0, 100);
      g_motion_goal.active = 0;
      g_motion_goal.started = 0;
      g_motion_goal.completed = 0;
      g_motion_goal.ack_sent = 0;
      command_in_progress = false;
      SendACK("CMD_COMPLETE");
      break;

    case CMD_FORWARD_INDEFINITE:
      current_robot_state = ROBOT_STATE_MOVING_FORWARD;
      motor_cmd.cmd_type = MOTOR_CMD_FORWARD;
      motor_cmd.left_pwm = 3000;
      motor_cmd.right_pwm = 3000;
      motor_cmd.duration_ms = 0;
      osMessageQueuePut(xMotorQueue, &motor_cmd, 0, 100);
      g_motion_goal.active = 0;
      g_motion_goal.started = 0;
      g_motion_goal.completed = 0;
      g_motion_goal.ack_sent = 0;
      command_in_progress = false;
      SendACK("CMD_COMPLETE");
      break;

    case CMD_BACKWARD_INDEFINITE:
      current_robot_state = ROBOT_STATE_MOVING_BACKWARD;
      motor_cmd.cmd_type = MOTOR_CMD_BACKWARD;
      motor_cmd.left_pwm = 3000;
      motor_cmd.right_pwm = 3000;
      motor_cmd.duration_ms = 0;
      osMessageQueuePut(xMotorQueue, &motor_cmd, 0, 100);
      g_motion_goal.active = 0;
      g_motion_goal.started = 0;
      g_motion_goal.completed = 0;
      g_motion_goal.ack_sent = 0;
      command_in_progress = false;
      SendACK("CMD_COMPLETE");
      break;

    case CMD_TURN_LEFT_45:
      Car_Turn(-45.0f);
      command_in_progress = true;
      break;

    case CMD_TURN_RIGHT_45:
      Car_Turn(45.0f);
      command_in_progress = true;
      break;

    case CMD_TURN_LEFT_90:
      Car_Turn(-79.0f);
      command_in_progress = true;
      break;

    case CMD_TURN_RIGHT_90:
      Car_Turn(80.0f);
      command_in_progress = true;
      break;

    case CMD_TURN_LEFT_180:
      Car_Turn(-180.0f);
      command_in_progress = true;
      break;

    case CMD_TURN_RIGHT_180:
      Car_Turn(180.0f);
      command_in_progress = true;
      break;

    case CMD_TURN_LEFT_90_BACK:
      Car_Turn(-90.0f);
      g_motion_goal.turn_backward = 1;
      command_in_progress = true;
      break;

    case CMD_TURN_RIGHT_90_BACK:
      Car_Turn(90.0f);
      g_motion_goal.turn_backward = 1;
      command_in_progress = true;
      break;

    default:
      SendACK("INVALID_CMD");
      break;
  }
}

/**
 * @brief Send ACK message back to RPi
 */
void SendACK(const char* message) {
  char ack_buffer[32];

  snprintf(ack_buffer, sizeof(ack_buffer), "ACK:%s\r\n", message);
  HAL_UART_Transmit(&huart3, (uint8_t*)ack_buffer, strlen(ack_buffer), 100);
}

/**
 * @brief Handle emergency stop condition
 */
void HandleEmergencyStop(void) {
  MotorCommand_t emergency_cmd;

  current_robot_state = ROBOT_STATE_EMERGENCY_STOP;

  emergency_cmd.cmd_type = MOTOR_CMD_BRAKE;
  osMessageQueuePut(xMotorQueue, &emergency_cmd, 0, 0);

  SendACK("EMERGENCY_STOP");

  SharedData_SetEmergencyStop(false);
  current_robot_state = ROBOT_STATE_IDLE;
  SendACK("EMERGENCY_CLEARED");
}

/**
 * @brief UART receive complete callback
 */
void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart) {
  if (huart == &huart3) {
    char rx = (char)uart_rx_buffer[0];

    if (rx == '\r' || rx == '\n') {
      uart_rx_index = 0;
      memset(uart_cmd_buffer, 0, sizeof(uart_cmd_buffer));
    } else if (rx == 'S' || rx == 's' || rx == 'L' || rx == 'l' || rx == 'R' || rx == 'r') {
      s_uart_single_char = rx;
      s_uart_char_ready = true;
      uart_rx_index = 0;
      memset(uart_cmd_buffer, 0, sizeof(uart_cmd_buffer));
    } else {
      if (uart_rx_index < sizeof(uart_cmd_buffer)) {
        uart_cmd_buffer[uart_rx_index] = (uint8_t)rx;
        uart_rx_index++;
      } else {
        uart_rx_index = 0;
        memset(uart_cmd_buffer, 0, sizeof(uart_cmd_buffer));
      }
    }

    HAL_UART_Receive_IT(&huart3, &uart_rx_buffer[0], 1);
  }
}
