import cv2
import os
from tqdm import tqdm

# Path to your video folder
# video_folder = "/home/users/ntu/leey0227/scratch/mdp_v2/zoomed_data"
video_folder = "/home/users/ntu/leey0227/scratch/mdp_v2/video_data"
# video_folder = "/home/users/ntu/leey0227/scratch/mdp_v2/test_video"

# Folder to save frames
# output_folder = "/home/users/ntu/leey0227/scratch/mdp_v2/given_zoomed_images/train/images"
output_folder = "/home/users/ntu/leey0227/scratch/mdp_v2/given_images/train/images"
os.makedirs(output_folder, exist_ok=True)

# List all video files in the folder
video_files = [f for f in os.listdir(video_folder) if f.lower().endswith(('.mp4', '.mov', '.avi', '.mkv'))]

frame_count = 0

for video_file in video_files:
    video_path = os.path.join(video_folder, video_file)
    cap = cv2.VideoCapture(video_path)

    if not cap.isOpened():
        print(f"⚠️ Could not open {video_file}, skipping.")
        continue

    total_frames = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))

    print(f"Processing {video_file} ({total_frames} frames)...")

    with tqdm(total=total_frames, desc=f"{video_file}", unit="frame") as pbar:
        while True:
            ret, frame = cap.read()
            if not ret:
                break  # Video ended

            # Include video name in frame filename to avoid overwriting
            frame_filename = os.path.join(
                output_folder, f"{os.path.splitext(video_file)[0]}_frame_{frame_count:06d}.png"
            )
            cv2.imwrite(frame_filename, frame)

            frame_count += 1
            pbar.update(1)

    cap.release()

print(f"✅ Saved {frame_count} frames from {len(video_files)} videos to {output_folder}")
